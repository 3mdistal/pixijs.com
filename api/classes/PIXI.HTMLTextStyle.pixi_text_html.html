align​

Type: 'left' | 'center' | 'right' | 'justify'

Description: Alignment for multiline text, does not affect single line text.

breakWords​

Type: boolean

Description: Indicates if lines can be wrapped within words, it needs wordWrap to be set to true.

dropShadow​

Type: boolean

Description: Set a drop shadow for the text.

dropShadowAlpha​

Type: number

Description: Set alpha for the drop shadow.

dropShadowAngle​

Type: number

Description: Set a angle of the drop shadow.

dropShadowBlur​

Type: number

Description: Set a shadow blur radius.

dropShadowColor​

Type: number | string

Description: A fill style to be used on the dropshadow e.g., 'red', '#00FF00'.

dropShadowDistance​

Type: number

Description: Set a distance of the drop shadow.

fill​

Type: string | Array<string> | number | Array<number> | CanvasGradient | CanvasPattern

Description: A canvas fillstyle that will be used on the text e.g., 'red', '#00FF00'. Can be an array to create a gradient e.g., ['#000000','#FFFFFF']MDN

fillGradientStops​

Type: number[]

Description: If fill is an array of colours to create a gradient, this array can set the stop points (numbers between 0 and 1) for the color, overriding the default behaviour of evenly spacing them.

fillGradientType​

Type: PIXI.TEXT_GRADIENT

Description: If fill is an array of colours to create a gradient, this can change the type/direction of the gradient.

fontFamily​

Type: string | string[]

Description: The font family, can be a single font name, or a list of names where the first is the preferred font.

fontSize​

Type: number | string

Description: The font size (as a number it converts to px, but as a string, equivalents are '26px','20pt','160%' or '1.6em')

fontStyle​

Type: 'normal' | 'italic' | 'oblique'

Description: The font style.

fontVariant​

Type: 'normal' | 'small-caps'

Description: The font variant.

fontWeight​

Type: 'normal' | 'bold' | 'bolder' | 'lighter' | '100' | '200' | '300' | '400' | '500' | '600' | '700' | '800' | '900'

Description: The font weight.

leading​

Type: number

Description: The space between lines.

letterSpacing​

Type: number

Description: The amount of spacing between letters, default is 0.

lineHeight​

Type: number

Description: The line height, a number that represents the vertical space that a letter uses.

lineJoin​

Type: 'miter' | 'round' | 'bevel'

Description: The lineJoin property sets the type of corner created, it can resolve spiked text issues. Default is 'miter' (creates a sharp corner).

miterLimit​

Type: number

Description: The miter limit to use when using the 'miter' lineJoin mode. This can reduce or increase the spikiness of rendered text.

padding​

Type: number

Description: Occasionally some fonts are cropped. Adding some padding will prevent this from happening by adding padding to all sides of the text.

stroke​

Type: string | number

Description: A canvas fillstyle that will be used on the text stroke, e.g., 'blue', '#FCFF00'

strokeThickness​

Type: number

Default Value: 0

Description: A number that represents the thickness of the stroke.

stylesheet​

Type: string

Description: Internal stylesheet contents, useful for creating rules for rendering

textBaseline​

Type: 'alphabetic' | 'top' | 'hanging' | 'middle' | 'ideographic' | 'bottom'

Description: The baseline of the text that is rendered.

trim​

Type: boolean

Description: Trim transparent borders.

whiteSpace​

Type: 'normal' | 'pre' | 'pre-line'

Description: How newlines and spaces should be handled. Default is 'pre' (preserve, preserve).

| value | New lines | Spaces | | ---------- | --------- | -------- | | 'normal' | Collapse | Collapse | | 'pre' | Preserve | Preserve | | 'pre-line' | Preserve | Collapse |

wordWrap​

Type: boolean

Description: Indicates if word wrap should be used.

wordWrapWidth​

Type: number

Description: The width at which text will wrap, it needs wordWrap to be set to true.
